11-3 Trailing Commas

Array


 [
    'foo',
+   'baz',
    'bar',
-   'baz',
 ]
        

const arr = [
  1,
  2,
  3,
];
arr; // [1, 2, 3]
arr.length; // 3
      

const arr = [1, 2, 3, , ,];
arr.length; // 5

Objects


const object = {
  foo: 'bar',
  baz: 'qwerty',
  age: 42,
};
      

function f(p) {}
function f(p,) {}

(p) => {};
(p,) => {};
      

class C {
  one(a,) {}
  two(a, b,) {}
}

const obj = {
  one(a,) {},
  two(a, b,) {},
};

Function calls


f(p);
f(p,);

Math.max(10, 20);
Math.max(10, 20,);

Illegal trailing commas


function f(,) {} // SyntaxError: missing formal parameter
(,) => {};       // SyntaxError: expected expression, got ','
f(,)             // SyntaxError: expected expression, got ','

function f(...p,) {} // SyntaxError: parameter after rest parameter
(...p,) => {}        // SyntaxError: expected closing parenthesis, got ','

Trailing commas in destructuring


// array destructuring with trailing comma
[a, b,] = [1, 2];

// object destructuring with trailing comma
const o = {
  p: 42,
  q: true,
};
const { p, q, } = o;
      

const [a, ...b,] = [1, 2, 3];
// SyntaxError: rest element may not have a trailing comma

Trailing commas in JSON


JSON.parse('[1, 2, 3, 4, ]');
JSON.parse('{"foo" : 1, }');
// SyntaxError JSON.parse: unexpected character
// at line 1 column 14 of the JSON data
      

JSON.parse('[1, 2, 3, 4 ]');
JSON.parse('{"foo" : 1 }');

Trailing commas in named imports and named exports

Named imports


import {
  A,
  B,
  C,
} from 'D';

import { X, Y, Z, } from 'W';

import { A as B, C as D, E as F, } from 'Z';

Named exports


export {
  A,
  B,
  C,
};

export { A, B, C, };

export { A as B, C as D, E as F, };

Quantifier prefix


/x{2}/; // Exactly 2 occurrences of 'x'; equivalent to /xx/
/x{2,}/; // At least 2 occurrences of 'x'; equivalent to /xx+/
/x{2,4}/; // 2 to 4 occurrences of 'x'; equivalent to /xxx?x?/
