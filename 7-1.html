7-1 數值

7-1-1 二進位和八進位實字


console.log(0b000) // 0 
console.log(0b001) // 1
console.log(0b010) // 2
console.log(0b011) // 3
console.log(0b100) // 4
console.log(0b101) // 5
console.log(0b110) // 6
console.log(0b111) // 7
          

console.log(parseInt('01')) // 1
console.log(parseInt('012')) // 10
console.log(parseInt('012', 10)) // 12


console.log(parseInt ('100', 8)) // 64
         

console.log(0o001) // 1
console.log(0o010) // 8
console.log(0o100) // 64
                 

console.log(0x0ff) // 255
console.log(0xf00) // 3840
