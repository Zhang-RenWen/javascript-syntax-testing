7-1-2 Number.isNaN


Number.isNaN(123)
// false，整數不是 NaN
Number.isNaN(Infinity)
// false ，Infinity不是NaN
Number.isNaN('a hundred' )
// false , 'a hundred'不是 NaN
Number.isNaN(NaN)
// true , NaN 是 NaN
Number.isNaN('a hundred'/ 'two')
// true
      

isNaN('a hundred')
// true，因為 Number('a hundred')是 NaN
isNaN(new Date())
// false, 因為 Number(new Date()) 使用 Date#valueOf 會回傳一個 unix timestamp 數值
      

function isNumber(value){
  return typeof value === 'number' && !Number.isNaN(value)
}
      

isNumber(1)
// true
isNumber(Infinity)
// true
isNumber(NaN)
// false
isNumber('two")
// false
isNumber(new Date())
// false

7-1-3 Number.isFinite


isFinite(NaN)
// false
isFinite(Infinity)
// false
isFinite(-Infinity)
// false
isFinite(null)
// true ，因為 Number(null) 因為 null 為 0
isFinite(-13)
// true ，因為 Number(-13) 為 -13
isFinite('10')
// true ，因為 Number(10) 為 10 
      

Number.isFinite(NaN)
// false
Number.isFinite(Infinity)
// false
Number.isFinite(-Infinity)
// false
Number.isFinite(null)
// false ，因為 Number(null) 因為 null 不是一個數值
Number.isFinite(-13)
// true，因為 Number(-13)
Number.isFinite('10')
// false ，因為 Number('10') 因為 '10' 不是一個數值
      

function numberIsFinite(value) {
  return typeof value ==='number' && isFinite(value)
}

function numberIsInteger(value){
  return Number.isFinite(value) && value % 1 === 0
}
