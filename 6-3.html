const handler = {
  get(target, key) {
    invariant(key,'get')
    return Reflect.get(target, key)
  },
  set(target, key, value){
    invariant(key,'set')
    return Reflect.set(target, key, value)
  },
},
has(target, key) {
  if (key.startswith('_')) {
    return false
  }
  return Reflect.has(target, key)
}
function invariant (key, action) {
  if(key.startsWith('_')) {  
    throw new Error (`Can't ${ action } private "${ key}"property`)
  }
}
