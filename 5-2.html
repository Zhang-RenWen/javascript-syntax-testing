const map = new WeakMap()
map.set(Date.now, 'now')
map.set(1, 1) // TypeError
map.set(Symbol(), 2) // TypeError


const map = new WeakMap([
  [new Date(),'foo' ],
  [() => 'bar', 'baz']
])
        

const date = new Date ()
const map = new WeakMap([[date, 'foo'], [() =>'bar','baz']])
map.has(date) // true
map.get(date) // 'foo'
map.delete(date) 
map.has(date) // false
